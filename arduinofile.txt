#include <SPI.h>
#include <LoRa.h>

const float ALKALINITY = 50.0;

// Style State
enum Style { ALE, LAGER, RED_WINE, WHITE_WINE };
Style currentStyle;
bool styleSelected = false;

void setup() {
  Serial.begin(9600);
  while (!Serial);

  Serial.println("--- Fermentation Monitor Initializing ---");

  if (!LoRa.begin(915E6)) {
    Serial.println("LoRa init failed!");
    while (1);
  }

  // --- STYLE SELECTION INPUT BOX ---
  Serial.println("\n==================================");
  Serial.println("STEP 1: SELECT YOUR STYLE");
  Serial.println("1: Ale | 2: Lager | 3: Red | 4: White");
  Serial.println("==================================");

  while (!styleSelected) {
    if (Serial.available() > 0) {
      char choice = Serial.read();
      
      // Filter valid choices
      if (choice == '1') { currentStyle = ALE; styleSelected = true; }
      else if (choice == '2') { currentStyle = LAGER; styleSelected = true; }
      else if (choice == '3') { currentStyle = RED_WINE; styleSelected = true; }
      else if (choice == '4') { currentStyle = WHITE_WINE; styleSelected = true; }

      if (styleSelected) {
        Serial.print(">> Choice confirmed: "); Serial.println(choice);
        Serial.println(">> Now listening for sensor packets...\n");
      }
      
      // Clear any leftover newline/box characters
      while(Serial.available() > 0) Serial.read();
    }
  }
}

void loop() {
  int packetSize = LoRa.parsePacket();
  
  if (packetSize > 0) {
    int numFloats = packetSize / 4; 
    float sensorValues[numFloats];

    // Your working Decoder Logic
    LoRa.readBytes((uint8_t *)sensorValues, packetSize);

    float co2Val  = sensorValues[0];
    float tempVal = sensorValues[1];
    float humVal  = sensorValues[2];

    // Calculation
    float phLevel = 0;
    if (co2Val > 0) {
      phLevel = 6.35 + log10(ALKALINITY / (0.0301 * co2Val));
    }

    // Display Data with your specific Warning Logic
    displayDashboard(co2Val, phLevel, tempVal, humVal);
  }
}

void displayDashboard(float co2, float ph, float temp, float hum) {
  float minT, maxT;
  String sName;

  // Set limits based on selection
  switch (currentStyle) {
    case ALE:        minT = 20.0; maxT = 22.0; sName = "ALE"; break;
    case LAGER:      minT = 7.0;  maxT = 13.0; sName = "LAGER"; break;
    case RED_WINE:   minT = 20.0; maxT = 30.0; sName = "RED WINE"; break;
    case WHITE_WINE: minT = 12.0; maxT = 20.0; sName = "WHITE WINE"; break;
  }

  Serial.println("--- NEW DATA RECEIVED ---");
  Serial.print("Style: "); Serial.println(sName);
  
  // Data Output
  Serial.print("CO2:  "); Serial.print(co2); Serial.println(" ppm");
  Serial.print("pH:   "); Serial.print(ph, 2); 
  if (ph < 3.2 && ph > 0.1) Serial.print(" [! CRITICAL]");
  Serial.println();

  Serial.print("Temp: "); Serial.print(temp); Serial.print(" C");
  if (temp < minT || temp > maxT) {
    Serial.print(" [! ALERT: Target "); Serial.print(minT); Serial.print("-"); Serial.print(maxT); Serial.print("]");
  }
  Serial.println();

  Serial.print("Hum:  "); Serial.print(hum); Serial.println(" %");
  Serial.print("RSSI: "); Serial.print(LoRa.packetRssi()); Serial.println(" dBm");
  Serial.println("-------------------------");
}
